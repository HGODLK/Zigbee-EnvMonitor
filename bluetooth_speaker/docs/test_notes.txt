#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>
#include <SPI.h>

#define TFT_CS   10
#define TFT_RST  8
#define TFT_DC   9
#define TFT_MOSI 11
#define TFT_SCLK 12
#define BLK_PIN  13

Adafruit_ST7735 tft = Adafruit_ST7735(&SPI, TFT_CS, TFT_DC, TFT_RST);

// 编码器引脚等（略）

void setup() {
  Serial.begin(115200);
  pinMode(BLK_PIN, OUTPUT);
  digitalWrite(BLK_PIN, HIGH);

  SPI.begin(TFT_SCLK, -1, TFT_MOSI, -1);   // 硬件 SPI 初始化
  tft.initR(INITR_BLACKTAB);
  tft.setRotation(1);
  tft.fillScreen(ST77XX_BLACK);
  // 后续绘图...
}
